16  package org.onehippo.forge.exdocpicker.api;
17  
18  import java.io.Serializable;
19  import java.util.Collection;
20  import java.util.Iterator;
21  
22  /**
23   * External document collection representation.
24   * {@link ExternalDocumentSearchService} is supposed to return an instance of this type.
25   *
26   * <P>
27   * Because it is used in a CMS/Wicket application, each item must be serializable and
28   * this collection itself must be serializable, too.
29   * </P>
30   *
31   * @param <T> Domain specific external document POJO type which must be serializable.
32   */
33  public interface ExternalDocumentCollection<T extends Serializable> extends Serializable, Cloneable {
34  
35      /**
36       * Returns an iterator of all the items.
37       * @return an iterator of all the items
38       */
39      Iterator<T> iterator();
40  
41      /**
42       * Returns an iterator of items starting from <code>first</code> index up to <code>count</code> items.
43       * @param first first index
44       * @param count iterating count
45       * @return an iterator of items starting from <code>first</code> index up to <code>count</code> items
46       */
47      Iterator<T> iterator(long first, final long count);
48  
49      /**
50       * Returns true if this collection contains the <code>doc</code>.
51       * @param doc document
52       * @return true if this collection contains the <code>doc</code>
53       */
54      boolean contains(T doc);
55  
56      /**
57       * Returns the index of {@code doc} in this collection if found.
58       * Otherwise returns -1.
59       * @param doc document
60       * @return the index of {@code doc} in this collection if found. Otherwise returns -1
61       */
62      int indexOf(T doc);
63  
64      /**
65       * Adds the <code>doc</code> to this collection.
66       * @param doc document
67       * @return true if added
68       */
69      boolean add(T doc);
70  
71      /**
72       * Inserts the <code>doc</code> at the {@code index} position of this collection.
73       * @param index index at which the document should be inserted
74       * @param doc document
75       * @return true if inserted
76       */
77      boolean add(int index, T doc);
78  
79      /**
80       * Adds all the <code>docs</code> to this collection.
81       * @param docs documents
82       * @return all the <code>docs</code> to this collection
83       */
84      boolean addAll(Collection<T> docs);
85  
86      /**
87       * Removes the <code>doc</code> from this collection.
88       * @param doc document
89       * @return the <code>doc</code> from this collection
90       */
91      boolean remove(T doc);
92  
93      /**
94       * Removes all the document items from this collection.
95       */
96      void clear();
97  
98      /**
99       * Returns the size of this collection.
100      * @return the size of this collection
101      */
102     int getSize();
103 
104     /**
105      * Copy all the items into an array and returns the array.
106      * @param a array
107      * @return all the items into an array and returns the array
108      */
109     T[] toArray(T[] a);
110 
111     /**
112      * Creates and returns a copy of this object.
113      * @return a copy of this object
114      */
115     Object clone();
116 
117 }
